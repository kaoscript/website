## Numbers

```ks
let decimal = 6

```

```ks
let time = 10_000_ms

let octal = 0o070

let radix = 36rNFfdH45
let float = 123_456.789_012
let exponent_positive = -7.2e+22
let exponent_negative = 1.1e-10
let float_exponent = 1.23456e-65
let float_dot = .789_012
```

## Arrays

```ks
let foo = [1, 2, 3]
```

```ks
let a = [1..5]
// 1, 2, 3, 4, 5

let b = [1..<5]
// 1, 2, 3, 4

let c = [1<..5]
// 2, 3, 4, 5

let d = [1<..<5]
// 2, 3, 4

let e = [1..6..2]
// 1, 3, 5

let f = [1<..<6..2]
// 3, 5

let g = [5..1]
// 5, 4, 3, 2, 1

let h = [5..1..2]
// 5, 3, 1

let i = [1..2..0.3]
// 1.0, 1.3, 1.6, 1.9
```

```ks
let min = 1
let max = 5

let a = [min..max]
// 1, 2, 3, 4, 5
```
