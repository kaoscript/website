# Variants

## type

```kaoscript
enum PersonKind {
    Director = 1
    Student
    Teacher
}

type SchoolPerson = {
    variant kind: PersonKind {
        Student {
            name: string
        }
    }
}

func greeting(person: SchoolPerson): String {
    return match person {
        .Teacher => "Hey Professor!"
        .Director => "Hello Director."
        .Student when .name == "Richard" => "Still here Ricky?"
        .Student => `Hey, \(.name).`
    }
}

func onlyStudent(student: SchoolPerson.Student) {
}

func onlyStudent(student: SchoolPerson(Student)) {
}

func onlyStudentOrTeacher(person: SchoolPerson(Student, Teacher)) {
}

echo(greeting({ kind: .Student, name: 'Richard' }))
echo(greeting({ kind: .Teacher }))

func testing(person: SchoolPerson) {
	if person is SchoolPerson.Student {
	}

	if person is .Student {
	}

	if person is SchoolPerson(Student) {
	}

	if person is SchoolPerson(Student, Teacher) {
	}
}
```
